Multiplier: 1x

let count = 0; let multiplier = 1; // Load saved data (so it remembers you) if (localStorage.getItem("count")) { count = parseInt(localStorage.getItem("count")); } if (localStorage.getItem("multiplier")) { multiplier = parseInt(localStorage.getItem("multiplier")); } const countSpan = document.getElementById("count"); const multiplierSpan = document.getElementById("multiplierDisplay"); function updateDisplay() { countSpan.textContent = count; multiplierSpan.textContent = multiplier; } // Click button document.getElementById("clickBtn").addEventListener("click", () => { count += multiplier; localStorage.setItem("count", count); updateDisplay(); }); // Set multiplier document.getElementById("setMultiplierBtn").addEventListener("click", () => { const value = parseInt(document.getElementById("multiplierInput").value); if (value >= 1) { multiplier = value; localStorage.setItem("multiplier", multiplier); updateDisplay(); } }); // Reset game document.getElementById("resetBtn").addEventListener("click", () => { count = 0; multiplier = 1; localStorage.clear(); updateDisplay(); }); updateDisplay();
